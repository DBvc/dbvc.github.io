# Practice 3.16

    (define (count-pairs x)
      (if (not (pair? x))
          0
          (+ (count-pairs (car x))
             (count-pairs (cdr x))
             1)))


![z1]({{site.url}}/static/images/3.16_1.png)

    (define z1 (cons (list 'a) (list 'b)))
    z1                                      ; ((a) b)
    (count-pairs z1)                        ; 3
